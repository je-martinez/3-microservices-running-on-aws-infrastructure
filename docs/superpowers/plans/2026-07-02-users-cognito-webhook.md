# Users Cognito Identity Webhook + Identity Tables — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Persist Cognito identity data into the Users service via a prod-faithful `POST /v1/webhooks/cognito` endpoint backed by two new Postgres tables, with a shared-secret auth guard and an env-gated local self-POST from the register flow.

**Architecture:** One Fastify endpoint and one persistence handler, identical in local and prod — only the trigger differs (prod Lambda shim vs. env-gated local self-POST). The webhook body is the **raw Cognito PostConfirmation event**; a Zod schema validates the fields we consume and preserves the whole event in `raw_payload`. Idempotency comes from `message_id = sha256(canonicalJSON(event))`. Every event is appended to `users_cognito_events` (never lost); when the `cognito_sub → user_id` link already exists, a 1:1 snapshot row in `users_cognito_data` is upserted last-write-wins.

**Tech Stack:** Node.js 24.18.0 (repo-pinned), Fastify, Prisma, Aurora Postgres, Zod, Vitest. Node's built-in `crypto` for the hash. No new runtime dependencies.

## Global Constraints

- **Node version:** run `nvm use` before any Node/pnpm command (repo pins 24.18.0 via `.nvmrc`).
- **Working directory for commands:** `services/users/` (the `@3mrai/users` pnpm package).
- **Code in English, converse in Spanish.** Code comments in English.
- **Conventions (referenced, never duplicated):** prefixed nano-ids (`ucd_`, `cge_`), full audit fields (`created_by/at`, `updated_by/at`, `deleted_by/at`), soft-delete, `snake_case` DB ↔ `camelCase` app via Prisma `@map`, all datetimes `TIMESTAMPTZ(6)`.
- **ESM imports:** source uses `.js` extensions on relative imports (e.g. `../../shared/id/nano-id.js`) — match the existing style exactly.
- **`users-impl` writes only source code** — never runs git or touches Linear; leaves work in the working tree for `github-ops`.
- **Data-model invariant:** the only hard FK is `users_cognito_data.user_id → users.id`. `users_cognito_events.cognito_sub` is an **indexed column, not an FK** — events may arrive before their snapshot row.
- **Response codes (webhook):** 200 known user, 202 unknown sub, 200 duplicate (no-op), 401 bad/missing secret, 422 invalid payload.
- **Test runner:** `pnpm test` (Vitest). Tests use manual mocks + `app.inject` for routes — no real DB.

```typescript
import { createHash } from "node:crypto";
import { z } from "zod";

// Raw AWS Cognito PostConfirmation event. We validate only the fields we
// consume; unknown fields pass through and are preserved in raw_payload.
export const cognitoEventSchema = z
  .object({
    triggerSource: z.string().min(1),
    callerContext: z.object({ clientId: z.string().optional() }).passthrough().optional(),
    request: z.object({
      userAttributes: z
        .object({
          sub: z.string().min(1),
          email: z.string().min(1),
        })
        .passthrough(),
    }),
  })
  .passthrough();

export type CognitoEvent = z.infer<typeof cognitoEventSchema>;

export function parseCognitoEvent(body: unknown): CognitoEvent {
  return cognitoEventSchema.parse(body);
}

export function extractFields(e: CognitoEvent): {
  cognitoSub: string;
  email: string;
  clientId: string | null;
  eventType: string;
} {
  return {
    cognitoSub: e.request.userAttributes.sub,
    email: e.request.userAttributes.email,
    clientId: e.callerContext?.clientId ?? null,
    eventType: e.triggerSource,
  };
}

// Stable serialization: recursively sort object keys so logically-equal
// events hash to the same value regardless of property order.
function canonicalize(value: unknown): unknown {
  if (Array.isArray(value)) return value.map(canonicalize);
  if (value && typeof value === "object") {
    return Object.keys(value as Record<string, unknown>)
      .sort()
      .reduce<Record<string, unknown>>((acc, key) => {
        acc[key] = canonicalize((value as Record<string, unknown>)[key]);
        return acc;
      }, {});
  }
  return value;
}

export function messageIdFor(e: CognitoEvent): string {
  return createHash("sha256").update(JSON.stringify(canonicalize(e))).digest("hex");
}
```

```typescript
import type { PrismaClient } from "@prisma/client";
import { newCognitoEventId } from "../../../shared/id/nano-id.js";
import type { CognitoEvent } from "./cognito-event.js";
import { extractFields, messageIdFor } from "./cognito-event.js";

export interface PersistCognitoDeps {
  writer: PrismaClient;
}

export interface PersistResult {
  linked: boolean;
}

export async function persistCognitoEvent(
  deps: PersistCognitoDeps,
  event: CognitoEvent,
): Promise<PersistResult> {
  const { cognitoSub, email, clientId, eventType } = extractFields(event);
  const messageId = messageIdFor(event);

  // 1. Append the event. Duplicate message_id → no-op (idempotent).
  try {
    await deps.writer.usersCognitoEvent.create({
      data: {
        id: newCognitoEventId(),
        cognitoSub,
        eventType,
        messageId,
        rawPayload: event as unknown as object,
      },
    });
  } catch (err: unknown) {
    if ((err as { code?: string }).code !== "P2002") throw err;
    // P2002 = unique violation on message_id: exact retry, already recorded.
  }

  // 2. Upsert the snapshot only when the register flow already linked the sub.
  const existing = await deps.writer.usersCognitoData.findUnique({ where: { cognitoSub } });
  if (!existing) {
    return { linked: false }; // → 202: event recorded, snapshot deferred.
  }

  await deps.writer.usersCognitoData.update({
    where: { cognitoSub },
    data: {
      email,
      clientId,
      lastEventType: eventType,
      rawPayload: event as unknown as object,
      updatedBy: cognitoSub,
    },
  });
  return { linked: true }; // → 200: event recorded + snapshot updated.
}
```

```typescript
import type { PrismaClient } from "@prisma/client";
import type { AuthProvider } from "../../../shared/auth/auth-provider.js";
import type { EventPublisher } from "../../../shared/messaging/event-publisher.js";
import { newUserId, newCognitoDataId } from "../../../shared/id/nano-id.js";
import { stampCreate } from "../../../shared/audit/audit.js";
import { toDomain, type User } from "../domain/user.js";

export interface RegisterConfig {
  localCognitoWebhook: boolean;
  webhookUrl: string;
  webhookSecret: string;
}

export type HttpPost = (
  url: string,
  body: unknown,
  headers: Record<string, string>,
) => Promise<{ status: number }>;

export interface RegisterDeps {
  writer: PrismaClient;
  auth: AuthProvider;
  events: EventPublisher;
  config: RegisterConfig;
  httpPost: HttpPost;
}

export interface RegisterInput {
  email: string;
  password: string;
  fullName: string;
  address?: unknown;
  phoneNumber?: string;
  e2eSource: boolean;
}

export async function registerUser(deps: RegisterDeps, input: RegisterInput): Promise<User> {
  const { sub } = await deps.auth.signUp(input.email, input.password);
  const id = newUserId();
  const tags = input.e2eSource ? ["E2E Source"] : [];
  const row = await deps.writer.user.create({
    data: {
      id,
      email: input.email,
      fullName: input.fullName,
      address: (input.address as any) ?? null,
      phoneNumber: input.phoneNumber ?? null,
      tags,
      ...stampCreate(id),
    },
  });

  // Establish the cognito_sub ↔ usr_… link. The webhook upserts this row later.
  await deps.writer.usersCognitoData.create({
    data: {
      id: newCognitoDataId(),
      userId: id,
      cognitoSub: sub,
      email: input.email,
      ...stampCreate(id),
      rawPayload: {},
    },
  });

  await deps.events.publishUserCreated({ id, email: input.email });

  // LOCAL only: the Cognito PostConfirmation trigger does not fire in the
  // emulator, so replay the same event to our own webhook. In prod the shim
  // does this and the flag is off.
  if (deps.config.localCognitoWebhook) {
    const event = {
      triggerSource: "PostConfirmation_ConfirmSignUp",
      callerContext: { clientId: "" },
      request: { userAttributes: { sub, email: input.email } },
    };
    await deps.httpPost(deps.config.webhookUrl, event, {
      "content-type": "application/json",
      "x-webhook-secret": deps.config.webhookSecret,
    });
  }

  return toDomain(row as any);
}
```

Note on `rawPayload: {}` in the link row: the register creates the snapshot with an empty payload; the webhook's last-write-wins upsert fills `rawPayload`/`clientId`/`lastEventType` on the first real event. This is consistent with the spec's upsert semantics (register owns the link; the webhook owns the Cognito-sourced fields).

```typescript
import { CognitoIdentityProviderClient } from "@aws-sdk/client-cognito-identity-provider";
import { env } from "../config/env.js";
import { writer, reader } from "../db/prisma.js";
import { NoopEventPublisher } from "../messaging/event-publisher.js";
import { CognitoAuthProvider } from "../auth/cognito-auth-provider.js";
import { registerUser } from "../../features/users/commands/register.js";
import { loginUser } from "../../features/users/commands/login.js";
import { updateProfile } from "../../features/users/commands/update-profile.js";
import { getMe } from "../../features/users/queries/get-me.js";
import { softDeleteE2EUsers } from "../../features/users/http/e2e-cleanup.js";
import { persistCognitoEvent } from "../../features/users/webhooks/persist-cognito-event.js";
import type { AppDeps } from "../../features/users/http/routes.js";

// Minimal fetch-based POST used by the register self-POST in local mode.
async function httpPost(url: string, body: unknown, headers: Record<string, string>) {
  const res = await fetch(url, { method: "POST", headers, body: JSON.stringify(body) });
  return { status: res.status };
}

export function buildContainer(): AppDeps {
  const cognito = new CognitoIdentityProviderClient({
    region: env.AWS_REGION,
    endpoint: env.AWS_ENDPOINT_URL,
  });
  const auth = new CognitoAuthProvider(cognito, env.COGNITO_USER_POOL_ID, env.COGNITO_CLIENT_ID);
  const events = new NoopEventPublisher();
  const registerConfig = {
    localCognitoWebhook: env.LOCAL_COGNITO_WEBHOOK,
    webhookUrl: env.COGNITO_WEBHOOK_URL,
    webhookSecret: env.COGNITO_WEBHOOK_SECRET,
  };
  return {
    env,
    webhookSecret: env.COGNITO_WEBHOOK_SECRET,
    registerUser: (_d, input) =>
      registerUser({ writer, auth, events, config: registerConfig, httpPost }, input as any),
    loginUser: (_d, input) => loginUser({ auth }, input),
    getMe: (_d, userId) => getMe({ reader }, userId),
    updateProfile: (_d, userId, input) => updateProfile({ writer }, userId, input as any),
    softDeleteE2EUsers: () => softDeleteE2EUsers({ writer }),
    persistCognitoEvent: (_d, event) => persistCognitoEvent({ writer }, event as any),
  };
}
```
